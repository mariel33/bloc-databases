1. adding: INSERT INTO
updating: UPDATE
deleting: DELETE

2. 
INSERT INTO: defines a set of rows that will be inserted into the table. You can insert multiple or single rows.
INSERT INTO (table name) (column names);

UPDATE: will update data using SET and WHERE. Can update multiple values or just one.
UPDATE (table name) SET (column name)=(new value) WHERE (column name)=(value);

DELETE: Will delete whichever data is specified. It is irreversible.
DELETE FROM (table name) WHERE (column name)=(value);

3. Data types to be used in a table:
Text- a person's name
Date- their date of birth
Numeric- year of graduation

CREATE TABLE spacecrafts (
    id varchar,
    name text,
    year launched integer,
    country of origin text,
    brief description text,
    orbiting body text,
    currently operating boolean,
    miles from earth varchar
);

INSERT INTO spacecrafts 
VALUES
('1966-027A', 'Luna 10', 1966, 'Soviet Union', 'First artificial satellite of the moon', 'Moon', 'N', '235000'),
('2010-020D', 'Akatsuki', 2010, 'Japan', 'A Japanese space mission to Venus', 'Venus', Y, '25 Million'),
('2013-063A', 'Maven', 2013, 'USA', 'Developed by NASA to study the Martian atmosphere while orbiting Mars', 'Mars', Y, '123 Million');

DELETE FROM spacecrafts WHERE id='1966-027A';

CREATE TABLE inbox (
    id integer,
    subject line text,
    sender text,
    additional recipients text SET DEFAULT 'None',
    email body text,
    time timestamp,
    read boolean,
    email chain id integer
);

INSERT INTO inbox

(0100, 'Meeting at 10', 'Jon Smith', 'None', 'Can you come to a meeting at 10?', 2018-03-13 09:06:00, N, 123),
